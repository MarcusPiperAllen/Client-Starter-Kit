// Walk fields in priority order; add a question for each that is empty in intent.
// If suggestions contain architectural signals (auth, payments, roles) and a slot
// remains, surface the top matching suggestion as a boolean question.
function buildFallbackQuestions(
  intent: ExtractedIntent,
  suggestions: string[],
): ResolutionQuestion[] {
  const PRIORITY_ORDER: Array<keyof ExtractedIntent> = [
    "organizationType", "technologyStack", "primaryGoal",
    "businessName", "tone", "callToAction", "audience", "contactEmail",
  ];

  const questions: ResolutionQuestion[] = [];

  for (const field of PRIORITY_ORDER) {
    if (questions.length >= 3) break;
    if (isFilled(intent, field)) continue;
    const template = FIELD_QUESTION_MAP[field];
    if (!template) continue;
    questions.push({ id: `q_${field}`, ...template });
  }

  // Fill remaining slots from architectural suggestions.
  if (questions.length < 3) {
    const archSignals = ["auth", "login", "payment", "pricing", "role", "account", "database"];
    for (const suggestion of suggestions.slice(0, 3)) {
      if (questions.length >= 3) break;
      if (archSignals.some((s) => suggestion.toLowerCase().includes(s))) {
        questions.push({
          id:           `q_suggestion_${questions.length}`,
          questionText: suggestion,
          rationale:    "This architectural decision affects the build prompt and starter code scaffolding.",
          answerType:   "boolean",
          options:      ["Yes, include this", "No, skip for now"],
          targetFields: ["notes"],
          priority:     "architectural",
        });
      }
    }
  }

  return questions;
}

const QUESTIONS_SYSTEM_PROMPT = `\
You are a project intake strategist for a web agency builder tool.
You receive a partially-filled project brief and identify the 1–3 most impactful gaps
to resolve before a developer can build the project.

Rank by impact:
  1. ARCHITECTURAL (highest): authentication, user accounts, payments, pricing tiers,
     user roles, database needs, external API integrations, MVP scope decisions.
  2. CONTENT (lower): missing business name, contact info, tone, CTA wording.

Rules:
- NEVER ask about a field already filled in the intent JSON.
- Cap at 3 questions. Prefer fewer high-impact questions over padding with low-value ones.
- For constrained fields, include the allowed options exactly as listed below.
- targetFields must be valid ExtractedIntent keys:
    projectName, businessName, founderName, organizationType, primaryGoal,
    audience, services, pages, tone, callToAction, technologyStack,
    contactEmail, contactPhone, notes
- Boolean questions (architectural decisions) use options ["Yes, include this", "No, skip for now"]
  and targetFields: ["notes"] — the answer will be appended to notes.

Allowed values for constrained fields:
  organizationType: ${ORG_TYPES.join(", ")}
  tone:             ${TONES.join(", ")}
  callToAction:     ${CTAS.join(", ")}
  technologyStack:  ${TECH_STACKS.join(", ")}

router.post(
  "/questions",
  async (req: Request, res: Response): Promise<void> => {
    // Validate request body against the generated Zod schema.
    const parseResult = ResolveQuestionsRequest.safeParse(req.body);
    if (!parseResult.success) {
      res.status(400).json({ error: "Invalid request body", issues: parseResult.error.issues });
      return;
    }
    const { intent, projectKind, suggestions, missingFields } = parseResult.data;

    let questions: ResolutionQuestion[];
    let source: "ai" | "fallback" = "ai";

    try {
      const userMessage = JSON.stringify(
        { intent, projectKind, suggestions, missingFields },
        null,
        2,
      );

      const response = await anthropic.messages.create({
        model:      MODEL_CONSTANT,   // <-- use the same model variable as the rest of the file
        max_tokens: 1024,
        system:     QUESTIONS_SYSTEM_PROMPT,
        messages:   [{ role: "user", content: userMessage }],
      });

      const text = response.content[0]?.type === "text" ? response.content[0].text : "";
      // Strip markdown code fences if the model wraps the JSON.
      const stripped = text.replace(/```(?:json)?/g, "").trim();
      const jsonMatch = stripped.match(/\{[\s\S]*\}/);
      if (!jsonMatch) throw new Error("No JSON object in AI response");

      const parsed = JSON.parse(jsonMatch[0]);
      const rawQuestions: unknown[] = Array.isArray(parsed.questions) ? parsed.questions : [];

      // Normalize; reject any question that is missing required fields.
      questions = rawQuestions
        .slice(0, 3)
        .map((q, i): ResolutionQuestion | null => {
          if (!q || typeof q !== "object") return null;
          const obj = q as Record<string, unknown>;
          const answerType = asString(obj.answerType);
          const valid: RQAnswerType[] = ["boolean", "single-select", "multi-select", "short-text"];
          if (!valid.includes(answerType as RQAnswerType)) return null;
          const targetFields = asStringArray(obj.targetFields);
          if (!targetFields.length) return null;
          const questionText = asString(obj.questionText);
          if (!questionText) return null;
          return {
            id:           asString(obj.id) || `q_ai_${i}`,
            questionText,
            rationale:    asString(obj.rationale),
            answerType:   answerType as RQAnswerType,
            options:      Array.isArray(obj.options)
                            ? (obj.options as unknown[]).map(asString).filter(Boolean)
                            : undefined,
            targetFields,
            priority:     asString(obj.priority) === "architectural" ? "architectural" : "content",
          };
        })
        .filter((q): q is ResolutionQuestion => q !== null);

      if (questions.length === 0) throw new Error("AI returned no valid questions");
    } catch (err) {
      req.log.warn({ err }, "questions: AI failed — using deterministic fallback");
      questions = buildFallbackQuestions(intent, suggestions);
      source = "fallback";
    }

    const validated = ResolveQuestionsResponse.parse({ questions, source });
    res.json(validated);
  },
);

interface ResolutionStepProps {
  questions:  ResolutionQuestion[];
  onAnswer:   (id: string, targetFields: string[], value: string | string[] | boolean) => void;
  onSkip:     (id: string) => void;
  onSkipAll:  () => void;
}

function ResolutionStep({ questions, onAnswer, onSkip, onSkipAll }: ResolutionStepProps) {
  const [shortText,    setShortText]    = useState<Record<string, string>>({});
  const [multiSelect,  setMultiSelect]  = useState<Record<string, string[]>>({});

  return (
    <div className="space-y-3">
      {questions.map((q) => (
        <div
          key={q.id}
          className="rounded-lg border border-border bg-card p-3 space-y-2"
        >
          {/* Header row: question + skip */}
          <div className="flex items-start justify-between gap-2">
            <div className="space-y-0.5">
              <p className="text-sm font-medium leading-snug">{q.questionText}</p>
              <p className="text-xs text-muted-foreground">{q.rationale}</p>
            </div>
            <button
              type="button"
              onClick={() => onSkip(q.id)}
              className="shrink-0 text-xs text-muted-foreground underline underline-offset-2 hover:text-foreground"
            >
              Skip
            </button>
          </div>

          {/* boolean */}
          {q.answerType === "boolean" && (
            <div className="flex gap-2">
              {(q.options ?? ["Yes", "No"]).map((opt, i) => (
                <button
                  key={opt}
                  type="button"
                  onClick={() => onAnswer(q.id, q.targetFields, i === 0)}
                  className="rounded-md border border-border px-3 py-1 text-xs font-medium
                             hover:bg-accent hover:text-accent-foreground transition-colors"
                >
                  {opt}
                </button>
              ))}
            </div>
          )}

          {/* single-select */}
          {q.answerType === "single-select" && q.options && (
            <div className="flex flex-wrap gap-1.5">
              {q.options.map((opt) => (
                <button
                  key={opt}
                  type="button"
                  onClick={() => onAnswer(q.id, q.targetFields, opt)}
                  className="rounded-full border border-border px-2.5 py-0.5 text-xs
                             capitalize hover:bg-primary hover:text-primary-foreground
                             transition-colors"
                >
                  {opt}
                </button>
              ))}
            </div>
          )}

          {/* multi-select */}
          {q.answerType === "multi-select" && q.options && (
            <div className="space-y-2">
              <div className="flex flex-wrap gap-1.5">
                {q.options.map((opt) => {
                  const selected = (multiSelect[q.id] ?? []).includes(opt);
                  return (
                    <button
                      key={opt}
                      type="button"
                      onClick={() =>
                        setMultiSelect((prev) => {
                          const cur = prev[q.id] ?? [];
                          return {
                            ...prev,
                            [q.id]: selected ? cur.filter((v) => v !== opt) : [...cur, opt],
                          };
                        })
                      }
                      className={`rounded-full border px-2.5 py-0.5 text-xs capitalize
                        transition-colors ${selected
                          ? "border-primary bg-primary text-primary-foreground"
                          : "border-border hover:bg-accent"}`}
                    >
                      {opt}
                    </button>
                  );
                })}
              </div>
              {(multiSelect[q.id]?.length ?? 0) > 0 && (
                <button
                  type="button"
                  onClick={() => onAnswer(q.id, q.targetFields, multiSelect[q.id] ?? [])}
                  className="text-xs font-medium text-primary hover:underline"
                >
                  Confirm ({multiSelect[q.id]?.length ?? 0} selected)
                </button>
              )}
            </div>
          )}

          {/* short-text */}
          {q.answerType === "short-text" && (
            <div className="flex gap-2">
              <input
                type="text"
                value={shortText[q.id] ?? ""}
                onChange={(e) =>
                  setShortText((prev) => ({ ...prev, [q.id]: e.target.value }))
                }
                onKeyDown={(e) => {
                  if (e.key === "Enter" && shortText[q.id]?.trim()) {
                    onAnswer(q.id, q.targetFields, shortText[q.id].trim());
                  }
                }}
                placeholder="Type your answer…"
                className="flex-1 rounded-md border border-border bg-background px-2 py-1
                           text-xs focus:outline-none focus:ring-1 focus:ring-primary"
              />
              <button
                type="button"
                disabled={!shortText[q.id]?.trim()}
                onClick={() => {
                  if (shortText[q.id]?.trim()) {
                    onAnswer(q.id, q.targetFields, shortText[q.id].trim());
                  }
                }}
                className="rounded-md bg-primary px-2.5 py-1 text-xs font-medium
                           text-primary-foreground disabled:opacity-40"
              >
                Save

        </div>

      <button
