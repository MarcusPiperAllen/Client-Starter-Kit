// Signal-based tech stack suggestion.
// Only fills an EMPTY technologyStack — never overrides an explicit user choice.
// If notes or primaryGoal contain simplicity signals, lean toward lighter stacks.
// Org-type defaults handle the common case where no signals are present.
const SIMPLICITY_SIGNALS = [
  "no backend", "no server", "no database", "no code", "no developer",
  "simple", "basic", "beginner", "volunteers", "limited budget", "low budget",
  "just html", "one page", "single page",
];

const ORG_SIMPLE_STACKS: Partial<Record<string, string>> = {
  "nonprofit":                 "html-css-js",
  "church/faith organization": "html-css-js",
  "community project":         "html-css-js",
  "local service business":    "html-css-js",
  "restaurant/food":           "html-css-js",
  "personal brand":            "html-css-js",
};

function suggestTechStack(intent: ExtractedIntent): string {
  // Already filled — respect the explicit choice.
  if (intent.technologyStack) return intent.technologyStack;

  // Check for simplicity signals in the free-text fields.
  const corpus = `${intent.notes} ${intent.primaryGoal}`.toLowerCase();
  if (SIMPLICITY_SIGNALS.some((s) => corpus.includes(s))) return "html-css-js";

  // Org-type default for non-technical audiences.
  const orgDefault = ORG_SIMPLE_STACKS[intent.organizationType];
  if (orgDefault) return orgDefault;

  // Nothing matched — leave empty so the miner surfaces the tech stack question.
  return "";
}


// ── In the POST /intent handler, around line 200 ────────────────────────
// BEFORE this line:

// REPLACE with:
const rawIntent = normalizeIntent(rawResult.intent);
const intent: ExtractedIntent = {
  ...rawIntent,
  technologyStack: suggestTechStack(rawIntent),
};
No TypeScript error: The spread + override pattern is safe because technologyStack is a string in ExtractedIntent and suggestTechStack always returns a string.
